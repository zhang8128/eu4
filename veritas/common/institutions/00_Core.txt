animals = {
	penalty = 0.5
	bonus = {
		global_manpower_modifier = 0.25
		development_cost = -0.25
	}


	history = {
		is_year = 1000
		OR = {
			AND = { #If this is changed costs for Nation Designer may also need to be changed.
				OR = {
					continent = new_world
					continent = north_america
					continent = south_america
					continent = oceania
				}
				owner = {
					is_playing_custom_nation = yes
					OR = {
						technology_group = western
						technology_group = scandinavian_tech
						technology_group = byzantine
						technology_group = eastern
						technology_group = ottoman
						technology_group = muslim
						technology_group = indian
						technology_group = indopersian
						technology_group = chinese
						technology_group = niitaka

					}
				}
			}
			continent = europe
			continent = asia
			region = egypt_region
			region = maghreb_region
			region = sahel_region
			region = niger_region
			region = guinea_region
			region = horn_of_africa_region
			region = south_africa_region
			area = northern_swahili_coast_area
			area = mozambique_area
			area = maravi_area
			owner = { technology_group = high_american }
		}
	}
	
	# no need for events, as its there from the start.
	#can_start = {	is_year = 1000	}
	#start_chance = 100
	#on_start = institution_events.1

	can_embrace = {
		OR = {
			NOT = { has_climate = tropical }
			has_port = yes
		}
	}

	embracement_speed = {
		modifier = {
			factor = 0.6
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_any_friendly_coast_border_province_has_embraced_animals
				any_friendly_coast_border_province = {
					animals = 100
				}
			}
		}
		modifier = {
			factor = 0.1
			scale = yes
			potential = {
				NOT = {
					any_friendly_coast_border_province = {
						animals = 100
					}
				}
			}

				tooltip = NEIGHBOR_NOT_OWNED_ANIMALS
