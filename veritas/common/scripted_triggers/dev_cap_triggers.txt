base_tax_max = { #Province Scope
	OR = {
		AND = {
			is_capital = no
			NOT = { 
				owner = { adm_tech = 12 }
			}
			base_tax = 2
		}
		AND = {
			is_capital = no
			owner = { adm_tech = 12 }
			NOT = { 
				owner = { adm_tech = 20 }
			}
			base_tax = 3
		}
		AND = {
			is_capital = no
			owner = { adm_tech = 20 }
			NOT = { 
				owner = { adm_tech = 26 }
			}
			base_tax = 4
		}
		AND = {
			is_capital = no
			owner = { adm_tech = 26 }
			base_tax = 5
		}
		AND = {
			is_capital = yes
			NOT = { province_id = 151 }
			NOT = { 
				owner = { adm_tech = 22 }
			}
			base_tax = 5
		}
		AND = {
			is_capital = yes
			NOT = { province_id = 151 }
			owner = { adm_tech = 22 }
			base_tax = 8
		}
		AND = {
			is_capital = yes
			province_id = 151
			base_tax = 10
		}
	}
}
