country_decisions = {

	peru_nation = {
		major = yes

		potential = {
			colonial_peru = { owned_by = ROOT }
			is_random_new_world = no
			is_nomad = no
			OR = {
				is_former_colonial_nation = yes
				is_colonial_nation = yes
			}
			NOT = { exists = PEU }
			OR = {
				ai = no
				num_of_cities = 3	
			}
			NOT = { has_country_flag = changed_from_colonial_nation }
		}

		allow = {
			stability = 1
			is_subject = no
			is_at_war = no
			is_former_colonial_nation = yes
			custom_trigger_tooltip = {
				NOT = { has_regency_trigger = yes }
				tooltip = no_regency
			}
			capital_scope = {
				colonial_region = colonial_peru
				is_core = ROOT
			}
			custom_trigger_tooltip = {
				all_owned_province = {
					OR = {
						NOT = { colonial_region = colonial_peru }
						AND = {
							colonial_region = colonial_peru
							NOT = { nationalism = 1 }
						}
					}
				}
				tooltip = no_separatism_peru
			}
		}

		effect = {
			change_tag = PEU
			if = {
				limit = {
					is_part_of_hre = yes
					is_elector = no
				}
				every_owned_province = {
					limit = { is_part_of_hre = yes }
					set_in_empire = no
				}
			}
			if = {
				limit = { NOT = { government_rank = 3 } }
				set_government_rank = 3
			}
			colonial_peru = {
				limit = { NOT = { owned_by = ROOT } }
				add_permanent_claim = PEU
			}
			add_prestige = 25
			set_country_flag = changed_from_colonial_nation
		}

		ai_will_do = { factor = 1 }
	}
}
