		name = "flavor_sco.EVTOPTB1" #Make peace with the Douglas family

		add_national_stability_or_adm_power = yes
	}

					region = oceanea_region

			}

			}

				owned_by = SCO
			}
			coromandel_region = {
				owned_by = SCO
			}
			malaya_region = {
				owned_by = SCO
			}
			moluccas_region = {
				owned_by = SCO
			}
			indonesia_region = {
				owned_by = SCO
			}
			oceanea_region = {
				owned_by = SCO
			}
		}
	}
	
	mean_time_to_happen = {
		months = 2000
	}
	
	option = {
		name = "flavor_sco.EVTOPTA6" #We will exploit this!

			name = "portuguese_india_trade_co"
