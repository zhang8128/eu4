# Queen Elizabeth I (1533-1603)
country_event = {
	id = women_history_events.1
	title = women_history_events.1.t
	desc = women_history_events.1.d
	picture = VEF_wihgfx_FEMALE_RULER_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = ENG
			tag = GBR
		}
		is_year = 1533
		NOT = { is_year = 1603 }
		NOT = { has_global_flag = wih_elizabeth_eng_flag }
		religion = catholic
		is_female = no
		NOT = { has_regency_trigger = yes }
		has_heir = no
		is_lesser_in_union = no
		government = monarchy
		any_owned_province = {
			religion = anglican
			is_in_capital_area = yes
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			set_global_flag = wih_elizabeth_eng_flag
		}
	}
	
	option = {
		name = women_history_events.1.a
		change_religion = anglican
		define_ruler = {
			name = "Elizabeth"
			dynasty = "Tudor"
			fixed = yes
			adm = 6
			dip = 6
			mil = 5
			female = yes
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality = inspiring_leader_personality }
		}
	}
	option = {
		name = women_history_events.1.b
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		define_advisor = {
			female = yes
			type = statesman #spymaster
			skill = 2
			discount = yes
			name = "Elizabeth Tudor"
		}
	}
}

		is_year = 1500

		OR = {
			AND = {
				tag = SPA
				OR = {
					central_america_region = {
						has_discovered = ROOT
					}
					mexico_region = {
						has_discovered = ROOT
					}
				}
				capital_scope = {
					continent = europe
				}
			}
			any_owned_province = {
				OR = {
					region = central_america_region

				}

					capital_scope = {
						continent = europe
					}
				}
