#Preparations
country_event = {
	id = jeu_paume.1
	title = "jeu_paume.EVTNAME1"
	desc = "jeu_paume.EVTDESC1"
	picture = JEU_DE_PAUME_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
	  	add_country_modifier = { name = "preparing_jeu_de_paume" duration = -1 }
	}

	option = {
		name = "jeu_paume.EVTOPTA1"	#Invite a neighbor
		ai_chance = { factor = 100 }
		trigger = {
			any_neighbor_country = {
				NOT = { has_country_flag = jeu_de_paume_boycott }
				NOT = { has_country_modifier = jeu_de_paume_grouch }
				NOT = { has_country_flag = jeu_de_paume }
				NOT = { has_country_modifier = lost_jeu_de_paume }
				NOT = { has_country_modifier = won_jeu_de_paume }
				NOT = { has_country_modifier = glorious_jeu_de_paume }
				NOT = { has_country_modifier = festivating_jeu_de_paume }
				technology_group = western
				NOT = { government = siena_republic }
				NOT = { government = revolutionary_republic }
				NOT = { government = revolutionary_empire }
				NOT = { has_regency_trigger = yes }
			}
		}
		add_dip_power = -2
		add_prestige = 1
		random_neighbor_country = {
		    limit = {
			    NOT = { has_country_flag = jeu_de_paume }
				NOT = { has_country_flag = jeu_de_paume_boycott }
				NOT = { has_country_modifier = jeu_de_paume_grouch }
				NOT = { has_country_modifier = lost_jeu_de_paume }
				NOT = { has_country_modifier = won_jeu_de_paume }
				NOT = { has_country_modifier = glorious_jeu_de_paume }
				NOT = { has_country_modifier = festivating_jeu_de_paume }
				technology_group = western
				NOT = { government = siena_republic }
				NOT = { government = revolutionary_republic }
				NOT = { government = revolutionary_empire }
				NOT = { has_regency_trigger = yes }
			}
		    country_event = { id = jeu_paume.2 days = 2 }
		}
	}
	
	option = {
		name = "jeu_paume.EVTOPTB1"	#It will be a local torunament
		add_prestige = -1
		hidden_effect = {
		    country_event = { id = jeu_paume.24 days = 30 }
		}
	}

}
