		clr_country_flag = sensa_ball_no
		clr_country_flag = sensa_bucintoro
		clr_country_flag = sensa_fine_bankett
		clr_country_flag = sensa_scandal_ball
		clr_country_flag = sensa_ball_two
		clr_country_flag = sensa_decision
		
		every_country = {
			limit = { has_country_flag = rival_at_sensa }
			country_event = { id = festa_sensa.19 }
        }
		
		every_country = {
			limit = { has_country_flag = neighbour_at_sensa }
			country_event = { id = festa_sensa.19 }
        }
		
		custom_tooltip = doge_add
		change_variable = { which = "doge_power" value = 3 }
	}
	
	option = {
		name = "festa_sensa.EVTOPTB0"	#Glory to Saint Mark!

		add_prestige = 12
		add_republican_tradition = 10
		add_country_modifier = { name = "glorious_sensa" duration = 3650 }

			limit = { has_country_flag = rival_at_sensa }
			country_event = { id = festa_sensa.19 }
       }

			limit = { has_country_flag = neighbour_at_sensa }

	id = festa_sensa.13
	title = "festa_sensa.EVTNAME13"

	picture = VEF_FEAST_eventPicture
	
	trigger = {
		any_country = { has_country_modifier = festivating_sensa }
		OR = { 
		    has_country_flag = neighbour_at_sensa
			has_country_flag = rival_at_sensa
		}
		NOT = { has_country_flag = sensa_fine_bankett_guest }
	}
	
	mean_time_to_happen = {
		days = 7
	}
	
	option = {
		name = "festa_sensa.EVTOPTA13"	#I've never seen so much food
		trigger = {
			    any_country = { has_country_flag = sensa_much_food }
				NOT = { any_country = { has_country_flag = sensa_no_food } }
		}
		ai_chance = {
			factor = 70
			modifier = {
				factor = 0.5
				is_rival = ROOT
			}
		}
		set_country_flag = sensa_fine_bankett_guest
		add_prestige = 1
		every_country = {
			        limit = { has_country_modifier = festivating_sensa }
			country_event = { id = festa_sensa.14 }
			add_opinion = { who = PREV modifier = opinion_pleased }
        }

	}
	
	option = {
		name = "festa_sensa.EVTOPTB13"	#It could've been better
		trigger = { NOT = { any_country = { has_country_flag = sensa_much_food } } }
		set_country_flag = sensa_fine_bankett_guest
		every_country = {
			limit = { has_country_modifier = festivating_sensa }
