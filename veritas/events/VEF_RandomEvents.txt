# Scandal in the Court - random event 5017
country_event = {
	id = vef_random.50000
	title = "vef_random.50000.t"
	desc = "vef_random.50000.d"
	picture = VEF_COURT_eventPicture
	
	is_triggered_only = yes

	trigger = {
		num_of_royal_marriages = 2
		OR = {
			AND = {
				government = monarchy
				NOT = { government = steppe_horde }
				NOT = { legitimacy = 80 }
			}
			AND = {
				government = republic
				NOT = { republican_tradition = 60 }
			}
		}
		NOT = { has_country_modifier = court_scandal_timer }
	}

	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 0.5
			ruler_has_personality = careful_personality
		}
		modifier = {
			factor = 1.25
			OR = {
				ruler_has_personality = drunkard_personality
				ruler_has_personality = babbling_buffoon_personality
			}
		}
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = { 
				name = "court_scandal_timer"
				duration = 7300 # 20 Years
				hidden = yes
			}
		}
	}

	option = {
		name = "vef_random.50000.opta"		# Take the blame and move on.
		ai_chance = { factor = 45 }
		add_country_modifier = { name = "scandal_at_court" duration = 1095 }
	}
	option = {
		name = "vef_random.50000.optb"		# Bribe an advisor to take the blame.
		ai_chance = { factor = 55 }
		add_years_of_income = -0.2
		add_corruption = 5
	}
	option = {
		name = "personality_options.5017.c" #Don't Panic!
		ai_chance = { factor = 100 }
		highlight = yes
		trigger = {
			ruler_has_personality = calm_personality
		}
		required_personality = calm_personality
		add_prestige = -10
	}
	option = {
		name = "personality_options.5017.dd" #Attempt to lessen the impact
		ai_chance = { factor = 100 }
		highlight = yes
		trigger = {
			ruler_has_personality = silver_tongue_personality
		}
		required_personality = silver_tongue_personality
		add_dip_power = -25
	}
	option = {
		name = "personality_options.5017.e" #It was their fault all along.
		ai_chance = { factor = 100 }
		highlight = yes
		trigger = {
			ruler_has_personality = babbling_buffoon_personality
		}
		required_personality = babbling_buffoon_personality
		add_prestige = -5
	}
}

	title = "vef_random.50013.t"
	desc = "vef_random.50013.d"

	picture = VEF_SHIP_SAILING_eventPicture

	is_triggered_only = yes

	trigger = {
		dip_tech = 13
		is_at_war = no
		OR = {
			has_idea_group = offensive_ideas
			has_idea_group = defensive_ideas
			has_idea_group = quality_ideas
			has_idea_group = quantity_ideas
			has_idea_group = standing_army_ideas
			has_idea_group = mercenary_ideas
		}
		NOT = { has_idea_group = mm1_ideas }
		NOT = { has_idea_group = gf1_ideas }
		num_of_ports = 4
		NOT = { navy_size_percentage = 0.95 }
	}

	mean_time_to_happen = { days = 1 }

	# The Admirals are right
	option = {
		name = "vef_random.50013.opta"
		ai_chance = { factor = 5 }
		add_mil_power = -50
		if = {
			limit = { 
				OR = {
					NOT = { heavy_ship_fraction = 20 } 
					NOT = { num_of_heavy_ship = 1 }
				}
			}
			random_owned_province = {
				limit = { has_port = yes }
				add_unit_construction = {
					amount = 4
					type = galley
					cost = 2
					speed = 0.5
				}
			}
		}
		if = {
			limit = { 
				heavy_ship_fraction = 20 
				num_of_heavy_ship = 1
			}
			random_owned_province = {
				limit = { has_port = yes }
				add_unit_construction = {
					amount = 2
					type = heavy_ship
					cost = 2
					speed = 0.5
				}
			}
		}
	}

	# We know what we are doing!
	option = {
		name = "vef_random.50013.optb"
		ai_chance = { factor = 95 }
		add_navy_tradition = -20
		add_country_modifier = { name = "upset_admirals" duration = 365 }
		# Actually, we don't!
		if = {
			limit = { NOT = { mil = 3 } }
			add_prestige = -5
		}
	}
}
