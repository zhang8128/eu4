# A PROVINCE EVENT CANNOT FIRE ITSELF, SO DOUBLES OF SOME EVENTS EXIST INTENTIONALLY


#Autonomy can be set to 10,20,30,40,60; effects entire area and newly captured provinces in the
#same area will inherit the values.

#Garrisons
	#Light Garrison - Reduce defensiveness for tax boost
	#Normal
	#Heavy Garrison - Improve defenisveness for tax loss
	#Recruit Skirmishers - Improve hostile attrition for tax loss
	#Stockpile Supplies - Improve supply limit and garrison growth; decrease hostile attrition

#Taxation
	#Harsh - Sacrifice unrest and trade for tax
	#Standard - No modifiers
	#Light - Sacrifice tax for unrest and trade

#Manpower
	#Mass Levies - Manpower boost for unrest
	#Standard - No modifiers
	#Exemptions - Manpower reduction for unrest

#Culture
	#Genocide - Take serious development hit but culture will change quickly
	#Deportation - Displace the local development but culture will change in the next few years
	#Encourage Accepted - Encourage a non-accepted culture in another group to change to an accepted one of the same group
	#Promote Culture - Unrest in exchange for slow cultural progress

# Main Menu
province_event = {
	id = vef_realm.1
	title = "vef_realm.1.t"
	desc = "vef_realm.1.d"
	picture = VEF_IMPORTANT_STATUE_eventPicture
	
	is_triggered_only = yes #vef_realm.999
	
	immediate = {
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				if = {
					limit = { NOT = { is_year = 1370 }}
					custom_tooltip = area_display_1
					else = { custom_tooltip = area_display_2 }
				}
			}
		}
		if = {
			limit = { has_province_flag = hold_10a }
			custom_tooltip = realm_autonomy_10
		}
		if = {
			limit = { has_province_flag = hold_20a }
			custom_tooltip = realm_autonomy_20
		}
		if = {
			limit = { has_province_flag = hold_30a }
			custom_tooltip = realm_autonomy_30
		}
		if = {
			limit = { has_province_flag = hold_40a }
			custom_tooltip = realm_autonomy_40
		}
		if = {
			limit = { has_province_flag = hold_60a }
			custom_tooltip = realm_autonomy_60
		}
		if = {
			limit = { has_province_flag = noble_titles }
			custom_tooltip = noble_titles
		}
		if = {
			limit = { has_province_flag = selling_land }
			custom_tooltip = selling_land
		}
		if = {
			limit = { has_province_flag = seizing_land_noble }
			custom_tooltip = seizing_land_noble
		}
		if = {
			limit = { has_province_flag = seizing_land_citizen }
			custom_tooltip = seizing_land_citizen
		}		
		if = {
			limit = { has_province_modifier = rm_light_garrison }
			custom_tooltip = tt_light_garrison
		}
		if = {
			limit = { has_province_modifier = rm_heavy_garrison }
			custom_tooltip = tt_heavy_garrison
		}
		if = {
			limit = { has_province_modifier = rm_skirmishers }
			custom_tooltip = tt_skirmishers
		}
		if = {
			limit = { has_province_modifier = rm_supplies }
			custom_tooltip = tt_supplies
		}
		if = {
			limit = {
				owner = {
					any_owned_province = {
						area = ROOT
						OR = {
							has_building = light_serfdom
							has_building = moderate_serfdom
							has_building = heavy_serfdom
						}
					}
				}
			}
			custom_tooltip = tt_serfdom_present
		}
		if = {
			limit = { has_province_modifier = rm_increase_serfdom }
			custom_tooltip = tt_serfdom_rising
		}
		if = {
			limit = { has_province_modifier = rm_decrease_serfdom }
			custom_tooltip = tt_serfdom_lowering
		}
	}
	
	option = {	#Autonomy
		trigger = {
			is_overseas = no
			owner = { is_nomad = no }
		}
		name = "vef_realm.1.opta"
		province_event = { id = vef_realm.2 }
	}

	option = {	#Garrisons
		trigger = { owner = { is_nomad = no }}
		name = vef_realm.1.optb
		province_event = { id = vef_realm.4 }
	}
	
	option = { #Taxes
		trigger = { owner = { is_nomad = no }}
		name = "vef_realm.1.optc"
		province_event = { id = vef_realm.6 }
	}
	
	option = { #Manpower
		trigger = { owner = { is_nomad = no }}
		name = "vef_realm.1.optd"
		province_event = { id = vef_realm.8 }
	}
	
	option = {	#Land Management
		trigger = { owner = { is_nomad = no }}
		name = "vef_realm.1.opte"
		province_event = { id = vef_realm.10 }
	}

	option = {	#Culture Conversion
		trigger = {
			owner = {
				any_owned_province = {
					area = ROOT
					OR = {
						has_province_modifier = rm_encourage_accepted
						has_province_modifier = rm_promote_culture
						has_province_modifier = rm_deportation
						has_province_modifier = rm_genocide
						AND = {
							NOT = { has_owner_culture = yes }
							NOT = { has_owner_accepted_culture = yes }
							NOT = {
								owner = {
									culture_group = PREV
									OR = {
										AND = {
											has_dlc = "Common Sense"
											government_rank = 4
										}
										AND = {
											NOT = { has_dlc = "Common Sense" }
											total_development = 1050
										}
									}
								}
							}
						}
					}
				}
				is_nomad = no
			}
			OR = {
				AND = {
					NOT = { continent = north_america }
					NOT = { continent = south_america }
					NOT = { continent = new_world }
				}
				NOT = { owner = { capital_scope = { continent = europe }}}
			}
		}
		name = "vef_realm.1.optf"
		province_event = { id = vef_realm.12 }
	}
	
	option = { #Serfdom
		trigger = {
			NOT = { continent = north_america }
			NOT = { continent = south_america }
			NOT = { continent = new_world }
			owner = { is_nomad = no }
		}
		name = "vef_realm.1.optg"
		province_event = { id = vef_realm.14 }
		if = {
			limit = { NOT = { owner = { adm_tech = 21 }}}
			custom_tooltip = serfdom_tech_limit
		}
		if = {
			limit = { owner = { has_reform = french_monarchy }}
			custom_tooltip = serfdom_french_limit
		}
		if = {
			limit = {
				owner = { capital_scope = { region = iberia_region }}
				NOT = { owned_by = GRA }
				GRA = { 
					num_of_owned_provinces_with = {
						value = 1
						region = iberia_region
					}
				}
			}
			custom_tooltip = serfdom_iberia_limit
		}
	}
	
	option = {	#Nomad Priority
		trigger = {
			is_nomad = yes
			NOT = { has_province_flag = nom_priority_set }
		}
		name = "vef_realm.1.opth"
		owner = {
			every_owned_province = {
				limit = {
					area = ROOT
					check_variable = { which = nom_priority value = 1 }
				}
				change_variable = { which = nom_priority value = 5 }
				set_province_flag = nom_priority_set
			}
		}
	}
	
	option = {	#Nomad Priority
		trigger = {
			is_nomad = yes
			has_province_flag = nom_priority_set
		}
		name = "vef_realm.1.opti"
		owner = {
			every_owned_province = {
				limit = {
					area = ROOT
					check_variable = { which = nom_priority value = 1 }
				}
				change_variable = { which = nom_priority value = -5 }
				clr_province_flag = nom_priority_set
			}
		}
	}
	
	option = {	#Area Selection
		name = "vef_realm.1.opty"
		owner = { country_event = { id = vef_realm.999 }}
	}
	
	option = {	#Close
		name = "vef_realm.1.optz"
		owner = { clr_country_flag = realm_management_active }
	}
	
}

		}
		if = {
			limit = { has_province_modifier = rm_supplies }
			custom_tooltip = tt_supplies
		}
	}

	option = {	#Light Garrison
		trigger = {
			NOT = { has_province_modifier = rm_light_garrison }
			NOT = { has_province_modifier = rm_heavy_garrison }
		}
		name = "vef_realm.4.opta"
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				add_province_modifier = { name = rm_light_garrison duration = -1 }
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = {	#Heavy Garrison
		trigger = {
			NOT = { has_province_modifier = rm_heavy_garrison }
			NOT = { has_province_modifier = rm_light_garrison }
		}
		name = "vef_realm.4.optb"
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				add_province_modifier = { name = rm_heavy_garrison duration = -1 }
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = {	#Normal Garrison
		trigger = {
			OR = {
				has_province_modifier = rm_light_garrison
				has_province_modifier = rm_heavy_garrison
			}
		}
		name = "vef_realm.4.optc"
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				if = {
					limit = { has_province_modifier = rm_light_garrison }
					remove_province_modifier = rm_light_garrison
				}
				if = {
					limit = { has_province_modifier = rm_heavy_garrison }
					remove_province_modifier = rm_heavy_garrison
				}
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = { #Skirmishers
		trigger = { NOT = { has_province_modifier = rm_skirmishers }}
		name = "vef_realm.4.optd"
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				add_province_modifier = { name = rm_skirmishers duration = -1 }
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = { #Disband Skirmishers
		trigger = { has_province_modifier = rm_skirmishers }
		name = "vef_realm.4.opte"
		owner = {
			every_owned_province = {
				limit = {
					area = ROOT
					has_province_modifier = rm_skirmishers
				}
				remove_province_modifier = rm_skirmishers
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = { #Supplies
		trigger = { NOT = { has_province_modifier = rm_supplies }}
		name = "vef_realm.4.optf"
		owner = {
			every_owned_province = {
				limit = { area = ROOT }
				add_province_modifier = { name = rm_supplies duration = -1 }
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = { #Disband Skirmishers
		trigger = { has_province_modifier = rm_supplies }
		name = "vef_realm.4.optg"
		owner = {
			every_owned_province = {
				limit = {
					area = ROOT
					has_province_modifier = rm_supplies
				}
				remove_province_modifier = rm_supplies
			}
			add_corruption = 1
		}
		province_event = { id = vef_realm.5 }
	}
	
	option = { #Back to Menu
		name = "vef_realm.4.opth"
		province_event = { id = vef_realm.1 }
	}

				GRA = { 
					NOT = { 
						num_of_owned_provinces_with = {	
							value = 1	
							region = iberia_region
						}
					}
				}
			}
			owner = {
				any_owned_province = {
					area = ROOT
					NOT = { has_building = heavy_serfdom }
				}
			}
		}

			every_owned_province = {
